We prefer sales data over net data for contact fields:

```sql
WITH enriched AS (
    SELECT
        d."Meter Reading (MR) Doc. No."   AS "Meter Reading (MR) Doc. No.",
        d.HAUS,
        d.ANLAGE,
        d.ME_MA_ID,
        d.EQUNR,
        d.HOUSE_NUM1,
        d.ENTSCHEIDUNG,

        -- Prefer sales_master data; if absent, fall back to net_master
        COALESCE(s.TITLE,      n.TITLE)      AS TITLE,
        COALESCE(s.FORENAME,   n.FORENAME)   AS FORENAME,
        COALESCE(s.SURNAME,    n.SURNAME)    AS SURNAME,
        COALESCE(s.STREET,     n.STREET)     AS STREET,
        COALESCE(s.STREETNO,   n.STREETNO)   AS STREETNO,
        COALESCE(s.POST_CODE1, n.POST_CODE1) AS POST_CODE1,
        COALESCE(s.CITY1,      n.CITY1)      AS CITY1,
        COALESCE(s.CITY2,      n.CITY2)      AS CITY2,
        COALESCE(s.EMAIL,      n.EMAIL)      AS EMAIL

    FROM J09D_temp d
    LEFT JOIN EABLG e
        ON e.INSTALLATION = d.INSTALLATION
    LEFT JOIN net_master n
        ON n.EQUNR = d.EQUNR
    LEFT JOIN sales_master s
        ON s.EQUNR = d.EQUNR
)
SELECT

    HAUS,
    ANLAGE,
    ME_MA_ID,
    EQUNR,
    HOUSE_NUM1,

    STREETNO,
    POST_CODE1,
    CITY1,
    CITY2,
```

You want to prefer `sales_master` when both exist; otherwise use `net_master`. The easiest way is to join both and use `COALESCE` to pick the non-NULL from `sales_master` first:

```sql
DROP TABLE IF EXISTS J09B_enriched;

CREATE TABLE J09B_enriched AS
SELECT
    d."Meter Reading (MR) Doc. No."       AS "Meter Reading (MR) Doc. No.",
    d.HAUS                                AS HAUS,
    d.ANLAGE                              AS ANLAGE,
    d.ME_MA_ID                            AS ME_MA_ID,
    d.EQUNR                               AS EQUNR,
    d.HOUSE_NUM1                          AS HOUSE_NUM1,
    d.ENTSCHEIDUNG                        AS ENTSCHEIDUNG,

    -- Prefer sales_master, fall back to net_master
    COALESCE(sm.TITLE,     nm.TITLE)      AS TITLE,
    COALESCE(sm.FORENAME,  nm.FORENAME)   AS FORENAME,
    COALESCE(sm.SURNAME,   nm.SURNAME)    AS SURNAME,
    COALESCE(sm.STREET,    nm.STREET)     AS STREET,
    COALESCE(sm.STREETNO,  nm.STREETNO)   AS STREETNO,
    COALESCE(sm.POST_CODE1,nm.POST_CODE1) AS POST_CODE1,
    COALESCE(sm.CITY1,     nm.CITY1)      AS CITY1,
    COALESCE(sm.CITY2,     nm.CITY2)      AS CITY2,
    COALESCE(sm.EMAIL,     nm.EMAIL)      AS EMAIL

FROM J09B_temp d
LEFT JOIN EABLG e
    ON d.ANLAGE = e.INSTALLATION
LEFT JOIN net_master nm
    ON d.EQUNR = nm.EQUNR
LEFT JOIN sales_master sm
    ON d.EQUNR = sm.EQUNR;
```

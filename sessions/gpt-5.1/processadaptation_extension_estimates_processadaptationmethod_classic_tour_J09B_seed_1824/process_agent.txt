       FROM J09B_temp t
       LEFT JOIN eablg g ON g.Installation = t.ANLAGE
     ),
     contact_pref AS (
       SELECT
         b.*,
         s.TITLE   AS s_TITLE,   s.FORENAME AS s_FORENAME, s.SURNAME AS s_SURNAME,

         n.EMAIL   AS n_EMAIL
       FROM base b
       LEFT JOIN sales_master s ON s.EQUNR = b.EQUNR
       LEFT JOIN net_master   n ON n.EQUNR = b.EQUNR
     )
     SELECT
       MR_Doc_No AS "Meter Reading (MR) Doc. No.",
       HAUS,
       ANLAGE,
       ME_MA_ID,
       EQUNR,
       HOUSE_NUM1,
       ENTSCHEIDUNG,
       -- prefer sales data when present
       CASE WHEN s_TITLE    IS NOT NULL THEN s_TITLE    ELSE n_TITLE    END AS TITLE,
       CASE WHEN s_FORENAME IS NOT NULL THEN s_FORENAME ELSE n_FORENAME END AS FORENAME,
       CASE WHEN s_SURNAME  IS NOT NULL THEN s_SURNAME  ELSE n_SURNAME  END AS SURNAME,
       CASE WHEN s_STREET   IS NOT NULL THEN s_STREET   ELSE n_STREET   END AS STREET,
       CASE WHEN s_STREETNO IS NOT NULL THEN s_STREETNO ELSE n_STREETNO END AS STREETNO,
       CASE WHEN s_POST_CODE1 IS NOT NULL THEN s_POST_CODE1 ELSE n_POST_CODE1 END AS POST_CODE1,
       CASE WHEN s_CITY1    IS NOT NULL THEN s_CITY1    ELSE n_CITY1    END AS CITY1,
       CASE WHEN s_CITY2    IS NOT NULL THEN s_CITY2    ELSE n_CITY2    END AS CITY2,
       CASE WHEN s_EMAIL    IS NOT NULL THEN s_EMAIL    ELSE n_EMAIL    END AS EMAIL
     FROM contact_pref;

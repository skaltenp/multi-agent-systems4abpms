```sql
WITH final_result AS (
    SELECT
        d."Meter Reading (MR) Doc. No." AS "Meter Reading (MR) Doc. No.",
        d.HAUS,
        e.ANLAGE,
        d.ME_MA_ID,
        d.EQUNR,
        d.HOUSE_NUM1,
        d.ENTSCHEIDUNG,

        CASE WHEN s.EQUNR IS NOT NULL THEN s.TITLE      ELSE n.TITLE      END AS TITLE,
        CASE WHEN s.EQUNR IS NOT NULL THEN s.FORENAME   ELSE n.FORENAME   END AS FORENAME,
        CASE WHEN s.EQUNR IS NOT NULL THEN s.SURNAME    ELSE n.SURNAME    END AS SURNAME,
        CASE WHEN s.EQUNR IS NOT NULL THEN s.STREET     ELSE n.STREET     END AS STREET,
        CASE WHEN s.EQUNR IS NOT NULL THEN s.STREETNO   ELSE n.STREETNO   END AS STREETNO,
        CASE WHEN s.EQUNR IS NOT NULL THEN s.POST_CODE1 ELSE n.POST_CODE1 END AS POST_CODE1,
        CASE WHEN s.EQUNR IS NOT NULL THEN s.CITY1      ELSE n.CITY1      END AS CITY1,
```
